class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        unordered_set<int> cpnums;
        for(int i = 0; i < nums.size(); i ++){
            if(cpnums.find(nums[i]) != cpnums.end())
                return true;
            else{
                cpnums.insert(nums[i]);
            }
        }
        return false;
    }
};
