class Solution {
public:
    int climbStairs(int n) {
        // if(n == 1||0) return 1;
        // else return climbStairs(n - 1) + climbStairs(n - 2);//递归，时间复杂度高
        if (n == 0) return 0;
        else if (n == 1) return 1;
        else{
        int num[n+1];
        num[0] = 1;
        num[1] = 1;
        for (int i = 2; i <= n; i++){
            num[i] = num[i-1] + num[i-2];
        }
        return num[n];}
    }
};
