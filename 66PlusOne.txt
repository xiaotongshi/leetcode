class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        //考虑1进位，2首位是0
        int n = digits.size();
        for (int i = n-1; i >=0; i--){
            if(digits[i]==9) digits[i]=0;//末位进位
            else {
                digits[i]=digits[i]+1;
                return digits;}//不用进位
        }
        if (digits.front() == 0) digits.insert(digits.begin(), 1);
        return digits;
    }
};
