/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        ListNode *a = headA;
        ListNode *b = headB;
        while (a!=b){
            //如果两条链表长度不同，第二次循环交换head之后可以弥补
            //if the length of first list if x and the second is x+n;
            //They will at the same position after x+n steps
            a = a?a->next:headB;
            b = b?b->next:headA;
        }
        return a;
    }
};
