/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode* curr = head;
        ListNode* prev = NULL;
        ListNode* next = NULL;
        while (curr != 0){
            next = curr -> next; //保存指向下一个地址的指针
            curr -> next = prev; //令指向下一个地址的指针为上一个地址
            prev = curr; //保存现有地址作为上一个地址
            curr = next; //移动到下一个地址
        }
        head = prev;
        return head;
    }
};
